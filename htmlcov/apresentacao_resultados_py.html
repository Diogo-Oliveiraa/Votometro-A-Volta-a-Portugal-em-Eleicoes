60# === Função para normalizar nomes (acentos, espaços, minúsculas) === 

61def normalizar_nome(nome: str) -> str: 

62 """ 

63 Normaliza um nome, removendo acentuacao, convertendo para minusculas 

64 e removendo espacos em excesso nas extremidades. 

65 """ 

66 return unicodedata.normalize('NFKD', str(nome)).encode('ASCII', 

67 'ignore').decode('ASCII').lower().strip() 

68 

69# === Função para identificar se é "Fora de Portugal" === 

70def eh_fora_de_portugal(nome: str) -> bool: 

71 """Verifica se o nome corresponde a 'Fora de Portugal'.""" 

72 return normalizar_nome(nome) in ["fora de portugal", "fora portugal"]

74# === Função para mostrar resultados com gráfico === 

75def mostra_resultados(df_resultados: pd.DataFrame, titulo: str) -> pd.DataFrame: 

76 """ 

77 Mostra no Streamlit uma tabela com os resultados eleitorais e um grafico interativo

86 # Detetar colunas com votos dos partidos 

87 colunas_partidos = [col for col in df_resultados.columns if col not in colunas_fixas] 

88 

89 # Somatório de votos por partido 

90 votos_por_partido = df_resultados[colunas_partidos].sum() 

91 total_votos_validos = votos_por_partido.sum() 

92 

93 # Criar DataFrame para gráfico 

94 df_grafico = pd.DataFrame({ 

95 "Partido": votos_por_partido.index, 

96 "Votos": votos_por_partido.values 

97 }) 

98 

99 # Calcular percentagem de votos 

100 df_grafico["Percentagem"] = ( 

101 round(df_grafico["Votos"] / total_votos_validos * 100, 2) 

102 if total_votos_validos > 0 else 0 

103 ) 

104 df_grafico = df_grafico.sort_values(by="Votos", ascending=False) # Ordenar por votos 

105 

106 # Escolha do tipo de gráfico (barras, circular, linha) 

107 tipo_grafico = st.selectbox(

110 key=f"grafico_{titulo}" 

111 ) 

112 

113 # Criar gráfico de barras 

114 if tipo_grafico == "Barras": 

115 fig = px.bar(df_grafico, x="Partido", y="Votos", color="Partido", 

116 text="Percentagem", height=700, title=titulo) 

117 fig.update_traces(texttemplate='%{text}%', textposition='outside') 

118 fig.update_layout(uniformtext_minsize=8, uniformtext_mode='hide') 

119 

120 # Criar gráfico circular

124 fig.update_traces(textinfo='percent+label')

126 # Criar gráfico de linha 

127 else:

129 title=titulo, hover_data=["Percentagem"])
